[Mapper]
public static partial class MapperConfiguration

     # [ 属性映射 ]
     [MapProperty(source: nameof(T1Class.Q1), target: nameof(T2Class.Q2))]

#### **[ 按枚举名 ]**

```csharp
[Mapper(EnumMappingStrategy = EnumMappingStrategy.ByName, EnumMappingIgnoreCase = true, AutoUserMappings = false)]
public static partial class MapperConfiguration
{
     
     [MapProperty(source: nameof(T1Class.Q1), target: nameof(T2Class.Q2))]
     public static partial T2Class Map(T1Class @class);

     # 通过 [ 自动映射 ] 来映射枚举值
     [UserMapping]
     public static Q2 MapQ1toQ2(Q1 q1) => q1 switch
     {
          Q1.E1 => Q2.E5,
          Q1.E2 => Q2.E6,
          Q1.E3 => Q2.E7,
          _ => throw new ArgumentOutOfRangeException(nameof(q1), q1, null)
     };

}

T1Class t1Class = new()
{
     Name = "John Doe",
     Q1 = Q1.E2,
};

# Q2.E6
T2Class t2Class = MapperConfiguration.Map(t1Class);


```
