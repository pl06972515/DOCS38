public class MapperAttribute : Attribute
{

     /// <summary>
     /// [ 映射规则 ] Source → Target ( 名称匹配 )
     /// {
     ///      CaseSensitive,   -- 匹配 Target 属性 | 构造函数参数名称时 [ 区分大小写 ]
     ///      CaseInsensitive, -- 匹配 Target 属性 | 构造函数参数名称时 [ 忽略大小写 ]
     /// }
     /// 
     /// </summary>
     public PropertyNameMappingStrategy PropertyNameMappingStrategy { get; set; } = PropertyNameMappingStrategy.CaseSensitive;
     // [ 映射规则 ] 将 Source = Null 成员 → [ Target 非空成员 ] ( true: ArgumentNullException, false: 忽略 Source 空属性赋值 )
     public bool AllowNullPropertyAssignment { get; set; } = true;
     // [ 映射规则 ] 将 Source 引用成员 → Target 引用成员 [ 默认浅克隆 ]
     public bool UseDeepCloning { get; set; }
     /// <summary>
     /// [ 映射规则 ] 循环引用
     /// - False: 默认不启用引用跟踪 ( 提升性能 )
     /// - True:  需要映射的对象之间存在引用关系或循环引用时 ( 建议开启 )
     /// 
     /// </summary>        
     public bool UseReferenceHandling { get; set; }  

     /// <summary>
     /// [ 枚举映射 ]
     /// {
     ///      ByValue, -- 按枚举值映射
     ///      ByName,  -- 按枚举名映射
     /// }
     /// 
     /// </summary>
     public EnumMappingStrategy EnumMappingStrategy { get; set; } = EnumMappingStrategy.ByValue;
     /// <summary>
     /// [ 枚举映射 ] EnumMappingStrategy.ByName 生效 ( 枚举名映射规则 )
     /// {
     ///      MemberName：直接使用枚举成员的原始名称( 不做任何格式转换: ValueOne )
     ///      CamelCase：首字母小写, 其余单词首字母大写( 例如: valueOne )
     ///      PascalCase：每个单词首字母都大写( 例如：ValueOne )
     ///      SerializationEnumMemberAttribute: 使用 [EnumMember] 特性定义的值( 例如：[EnumMember("valueOne")] )
     /// }
     /// 
     /// </summary>
     public EnumNamingStrategy EnumNamingStrategy { get; set; } = EnumNamingStrategy.MemberName;
     /// [ 枚举映射 ] EnumMappingStrategy.ByName 生效 ( 匹配名称是否忽略大小写 )
     public bool EnumMappingIgnoreCase { get; set; }

     # [ 警告规则 ] 当 Source Generator 生成映射代码时( 即编译期间 )
     /// <summary>
     /// [Flags]
     /// public enum RequiredMappingStrategy
     /// {
     ///      None,   -- 不发出任何 [ 未映射成员的警告 ]
     ///      Both,   -- 针对 [ 未映射的源 ] + [ 目标成员 ] 都发出警告
     ///      Source, -- 仅对 [ 未映射的 Target 源成员 - 发出警告 ]
     ///      Target  -- 仅对 [ 未映射的 Source 目标成员 - 发出警告 ]
     /// }
     /// 
     /// </summary>
     public RequiredMappingStrategy RequiredMappingStrategy { get; set; } = RequiredMappingStrategy.Both;

     # [ 映射规则 ] 创建 Target 对象时 [ 构造函数选择策略 ]
     public bool PreferParameterlessConstructors { get; set; } = true;
    
     # [ 映射规则 ] 是否启用自动检测 - 类型映射( false: 需标记 [UserMapping] )
     public bool AutoUserMappings { get; set; } = true;

}
