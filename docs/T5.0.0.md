```csharp
[Mapper]
public static partial class MapperConfiguration
{

     // [MapProperty(source: nameof(T1Class), target: nameof(T2Class.Q2File), Use = nameof(Q1Map))]
     [MapPropertyFromSource(target: nameof(T2Class.Q2File), Use = nameof(Q1Map))]
     public static partial T2Class MapTo(T1Class @class);

     private static Q2 Q1Map(T1Class q1)
     {
          return new Q2(q1.QFile1, q1.QFile2);
     }

}
```
